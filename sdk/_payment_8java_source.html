672  public Payment(Parcel in, int parcelVersion) {
673  super(in, parcelVersion);
674  mPaymentId = in.readString();
675  mLocalPaymentId = in.readString();
676  mInvoice = in.readString();
677  mVenue = in.readString();
678  mLane = in.readString();
679  mCashierId = in.readString();
680  mPaymentType = ConversionUtility.readEnumFromParcel(in, PaymentType.class);
681  mPaymentMethod = in.readString();
682  mAuthorizationMethod = ConversionUtility.readEnumFromParcel(in, AuthorizationMethod.class);
683  mAuthResult = ConversionUtility.readEnumFromParcel(in, AuthorizationResult.class);
684  mAuthCode = in.readString();
685  mAuthResponseText = in.readString();
686  mCardInformation = in.readParcelable(CardInformation.class.getClassLoader());
687  mAppSpecificData = in.readString();
688  mRequestedPaymentAmount = ConversionUtility.readBigDecimalFromParcel(in);
689  mPaymentAmount = ConversionUtility.readBigDecimalFromParcel(in);
690  mGratuityAmount = ConversionUtility.readBigDecimalFromParcel(in);
691  mTaxAmount = ConversionUtility.readBigDecimalFromParcel(in);
692  mCashbackAmount = ConversionUtility.readBigDecimalFromParcel(in);
693  mRedeemedUnits = in.readString();
694  mReceipts = new HashMap<>(Receipt.TYPE_MAXIMUM + 1);
695  in.readMap(mReceipts, Receipt.class.getClassLoader());
696  mSignature = in.readParcelable(Bitmap.class.getClassLoader());
697  mTimestamp = in.readString();
698  if (getParcelVersion() >= 8) {
699  mRequestedCardPresentationMethods =
700  ConversionUtility.readEnumArrayFromParcel(
701  in, CardInformation.PresentationMethod.class);
702  }
703  }

735  @Override
736  public void writeToParcel(Parcel dest, int flags) {
737  super.writeToParcel(dest, flags);
738  dest.writeString(mPaymentId);
739  dest.writeString(mLocalPaymentId);
740  dest.writeString(mInvoice);
741  dest.writeString(mVenue);
742  dest.writeString(mLane);
743  dest.writeString(mCashierId);
744  if (getParcelVersion() < 9 && mPaymentType == PaymentType.MOBILE) {
745  ConversionUtility.writeEnumToParcel(dest, PaymentType.ALTERNATE);
746  } else {
747  ConversionUtility.writeEnumToParcel(dest, mPaymentType);
748  }
749  dest.writeString(mPaymentMethod);
750  ConversionUtility.writeEnumToParcel(dest, mAuthorizationMethod);
751  if (getParcelVersion() < 5) {
752  // The auth results before version 5 did not contain cash verified or cancelled externally,
753  // so convert them to something that is similar.
754  if (mAuthResult == AuthorizationResult.CASH_VERIFIED) {
755  mAuthResult = AuthorizationResult.AUTHORIZED_EXTERNALLY;
756  } else if (mAuthResult == AuthorizationResult.CANCELLED_EXTERNALLY) {
757  mAuthResult = AuthorizationResult.USER_CANCELLED;
758  }
759  }
760  if (getParcelVersion() < 9 && mAuthResult == AuthorizationResult.IN_PROGRESS) {
