62  public enum AuthorizationResult {
63  /** The payment was approved directly. */
64  AUTHORIZED_ONLINE,

66  AUTHORIZED_OFFLINE,
67  /** The payment was rejected directly. */
68  REJECTED_ONLINE,
69  /**
70  * The payment was above the configured threshold and it was unable to reach the

72  * at a later time by manual intervention. Refer to AUTHORIZED_EXTERNALLY for the proper
73  * status if approved later in this manner.
74  */
75  REJECTED_OFFLINE,
76  /** The user cancelled the payment. */
77  USER_CANCELLED,

84  * This represents a payment that was voided. Use {@link #getReferencePaymentId()} to get
85  * the original payment ID.
86  */
87  VOIDED,
