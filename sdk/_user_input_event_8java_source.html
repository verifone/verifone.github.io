700  /**
701  * Validates the current values for the given Input Type.
702  */
703  public boolean isValidForType(final String inputType) {
704  boolean isValid = false;
705  switch (inputType) {
706  case TransactionManager.INPUT_TYPE_EMAIL:
707  case TransactionManager.INPUT_TYPE_PASSWORD:
708  case TransactionManager.INPUT_TYPE_TEXT:
709  isValid = mValue != null;
710  break;
711  case TransactionManager.INPUT_TYPE_NUMBER:
712  case TransactionManager.INPUT_TYPE_DECIMAL:
713  isValid = mNumericValue != null;
714  break;
715  case TransactionManager.INPUT_TYPE_CONFIRMATION:
716  case TransactionManager.INPUT_TYPE_MANAGER_APPROVAL:
717  case TransactionManager.INPUT_TYPE_ANY_KEY:
718  isValid = mIsConfirmed != null;
719  break;
720  case TransactionManager.INPUT_TYPE_MENU_OPTIONS:
721  isValid = (mSelectedIndices != null && !mSelectedIndices.isEmpty());
722  break;
723  case TransactionManager.INPUT_TYPE_SIGNATURE:
724  isValid = mImage != null;
