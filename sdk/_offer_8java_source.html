223  public Offer(Parcel in, int parcelVersion) {
224  super(in, parcelVersion);
225  setOfferId(in.readString());
226  setOfferType(in.readInt());
227  setOfferDescription(in.readString());
228  setOfferRefundable(in.readInt() == 1);
229  setOfferCombinable(in.readInt() == 1);
230  setOfferPercentDiscount(ConversionUtility.readBigDecimalFromParcel(in));
231  setOfferDiscount(ConversionUtility.readBigDecimalFromParcel(in));
232  setProgramId(in.readString());
233  setMerchantOfferCode(in.readString());
234  setProductCode(in.readString());
235  setAssociatedProductCode(in.readString());
236  setSpecialProductCode(in.readString());
237  setQrCode(in.readString());
238  }

240  @Override
241  public void writeToParcel(Parcel dest, int flags) {
242  super.writeToParcel(dest, flags);
243  dest.writeString(getOfferId());
244  dest.writeInt(getOfferType());
245  dest.writeString(getOfferDescription());
246  dest.writeInt((getOfferRefundable() ? 1 : 0));
247  dest.writeInt((getOfferCombinable() ? 1 : 0));
248  ConversionUtility.writeBigDecimalToParcel(getOfferPercentDiscount(), dest);
249  ConversionUtility.writeBigDecimalToParcel(getOfferDiscount(), dest);
250  dest.writeString(getProgramId());
251  dest.writeString(getMerchantOfferCode());
252  dest.writeString(getProductCode());
253  dest.writeString(getAssociatedProductCode());
254  dest.writeString(getSpecialProductCode());
255  dest.writeString(getQrCode());
256  }
