329  public Offer(Parcel in, int parcelVersion) {
330  super(in, parcelVersion);
331  setOfferId(in.readString());
332  setOfferType(in.readInt());
333  if (parcelVersion < 6) {
334  // Set the description based off of the old offer description field.
335  setDescription(in.readString());
336  }
337  setOfferRefundable(in.readInt() == 1);
338  setOfferCombinable(in.readInt() == 1);
339  setOfferPercentDiscount(ConversionUtility.readBigDecimalFromParcel(in));
340  setOfferDiscount(ConversionUtility.readBigDecimalFromParcel(in));
341  setProgramId(in.readString());
342  setMerchantOfferCode(in.readString());
343  setProductCode(in.readString());
344  setAssociatedProductCode(in.readString());
345  setSpecialProductCode(in.readString());

349  @Override

387  @NonNull
388  @Override
389  public JSONObject buildToCpJson() {
390  JSONObject jsonObject = new JSONObject();
391  try {
392  jsonObject.put("Offer_Id", getOfferId());
393  jsonObject.put("Program_Id", getProgramId());
394  OfferType offerType = OfferType.findById(getOfferType());
395  jsonObject.put("Offer_Type", offerType != null ? offerType.name() : null);
396  jsonObject.put("Offer_Description", getDescription());
397  jsonObject.put("Offer_Refundable", getOfferRefundable());
398  jsonObject.put("Offer_Combinable", getOfferCombinable());
399  jsonObject.put("Offer_Percent_Discount", getOfferPercentDiscount());
400  jsonObject.put("Merchant_Offer_Code", getMerchantOfferCode());
401  jsonObject.put("Product_Code", getProductCode());
402  jsonObject.put("Special_Product_Code", getSpecialProductCode());
403  } catch (JSONException e) {
404  Log.w(TAG, "SDK Unable to put value into this object. " + e.getMessage());
405  }
