140  /**
141  * Requests a mode change by sending an appropriate targeted broadcast.
142  * @param context Used to fire the broadcast.
143  * @param mode The specific mode to enable.
144  * @param options (optional) Options such as the color scheme or interface type.
145  * @return The result of sending the broadcast.
146  */
147  public static void requestMode(@NonNull final Context context,
148  @Nullable final String mode,
149  @Nullable final String... options) {
150  Log.initialize((Application) context.getApplicationContext());
151  Log.i(TAG, "requestMode " + mode + " from " + context.getPackageName() + ".");
152  Intent intent = new Intent(ACTION_REQUEST_ACCESSIBILITY_CHANGE);
153  intent.addCategory(CATEGORY_ACCESSIBILITY);
154  intent.putExtra(EXTRA_MODE, mode);
155  if (options != null && options.length > 0) {
156  for (String option : options) {
157  boolean hasRecognizedOption = false;
158  switch (mode) {
159  case MODE_ASSISTANCE: {
160  switch (option) {
161  case COLOR_SCHEME_USER_SELECT:
162  hasRecognizedOption = true;
163  intent.putExtra(EXTRA_COLOR_SCHEME, option);
164  break;
165  case AMOUNT_VOCALIZATION_ENABLED:
166  case AMOUNT_VOCALIZATION_DISABLED:
167  /*Sets the vocalization button state value(option) through navigator_settings*/
168  hasRecognizedOption = true;
169  intent.putExtra(EXTRA_AMOUNT_VOCALIZATION, option);
170  break;
171  default: // switch option default
172  break;
173  }
174  break;
175  }
176  case MODE_TRAINING: {
177  switch (option) {
178  case MODE_EXTENSION_EXPLORATION:
179  case MODE_EXTENSION_DOUBLE_TAP:
180  case MODE_EXTENSION_TRAIN_EDGES:
181  case MODE_EXTENSION_NAVIGATE:
182  case MODE_EXTENSION_SUBMIT_DIGIT:
183  case MODE_EXTENSION_ENTER_PIN:
184  hasRecognizedOption = true;
185  intent.putExtra(EXTRA_MODE_EXTENSION, option);
186  break;
187  default: // switch option default
188  break;
189  }
190  break;
191  }
192  default: // switch mode default
193  /*Sets the vocalization button state value(option) through navigator_settings, when no mode is specified*/
194  switch (option) {
195  case AMOUNT_VOCALIZATION_ENABLED:
196  case AMOUNT_VOCALIZATION_DISABLED:
197  hasRecognizedOption = true;
198  intent.putExtra(EXTRA_AMOUNT_VOCALIZATION, option);
199  break;
200  default: break;
201  }
202  break;
203  }
204  if (!hasRecognizedOption) {
205  Log.w(TAG, "Unable to set option " + option + " for mode " + mode + ".");
206  }
207  }
208  }
209  context.sendBroadcast(intent, PERMISSION_REQUEST_ACCESSIBILITY_CHANGE_RECEIVER);
210  }

212  /**
213  * Convenient method to register a receiver for the {@link #ACTION_ACCESSIBILITY_CHANGED}
214  * broadcast. The receiver is registered using the provided context. Refer to
215  * {@link Context#registerReceiver(BroadcastReceiver, IntentFilter, String, Handler)} for more
216  * information.
217  */
218  public static Intent registerReceiver(@NonNull final Context context,
219  @NonNull BroadcastReceiver receiver,
220  @Nullable Handler scheduler) {
221  Log.initialize((Application) context.getApplicationContext());
222  Log.i(TAG, "registerReceiver from " + context.getPackageName() + ".");
223  IntentFilter filter = new IntentFilter(ACTION_ACCESSIBILITY_CHANGED);
224  filter.addCategory(CATEGORY_ACCESSIBILITY);
225  return context.registerReceiver(receiver, filter,
226  PERMISSION_REQUEST_ACCESSIBILITY_CHANGE_RECEIVER,
227  scheduler);
228  }
