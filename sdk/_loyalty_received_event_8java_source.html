106  * Sets the offers and performs the calculations to change the appropriate amounts.
107  * Please note, this does not change the tax amount, only the total amounts for both the
108  * transaction and the current payment.
109  */
110  public void applyLoyaltyOffers(ArrayList<Offer> loyaltyOffers) {
111  setLoyaltyOffers(loyaltyOffers);
112  if (loyaltyOffers != null && !loyaltyOffers.isEmpty()) {
113  final BigDecimal originalTransactionAmount = getTransaction().getAmount();
114  BigDecimal newTransactionAmount = getTransaction().getAmount();
115  final BigDecimal originalAmount = getResponsePayment().getRequestedPaymentAmount();
116  BigDecimal newRequestedAmount = getResponsePayment().getRequestedPaymentAmount();
117  for (Offer offer : loyaltyOffers) {
118  BigDecimal transactionAugend;
119  BigDecimal paymentAugend;
120 
121  if (offer.getOfferDiscount() != null) {
122  transactionAugend = offer.getOfferDiscount();
123  paymentAugend = offer.getOfferDiscount();
124  } else if (offer.getOfferPercentDiscount() != null) {
125  transactionAugend = originalTransactionAmount.multiply(
126  offer.getOfferPercentDiscount());
127  paymentAugend = originalAmount.multiply(
128  offer.getOfferPercentDiscount());
129  } else {
130  transactionAugend = offer.getAmount();
131  paymentAugend = offer.getAmount();
132  }
133  if (transactionAugend != null) {
134  newTransactionAmount = newTransactionAmount.add(transactionAugend);
135  }
136  if (paymentAugend != null) {
137  newRequestedAmount = newRequestedAmount.add(paymentAugend);
138  }
139  }
140  getTransaction().setAmount(newTransactionAmount);
141  getResponsePayment().setRequestedPaymentAmount(newRequestedAmount);
142  }
143  }
