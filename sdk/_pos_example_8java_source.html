19 /**
20  * Example POS app for the Commerce Platform Payment API.
21  * In a typical flow, the session and basket will be opened, reserving the payment terminal to be able to display items on screen as they are added to the basket.
22  * The cashier will add items to the basket using the screen or a scanner, calling the {@link #addMerchandiseToBasket(Merchandise merch) addMerchandiseToBasket(Merchandise merch)} method,
23  * which adds the item to the basket and updates the display with the line item and the updated running total.
24  * Once the cashier has finished adding items, the {@link #basketFinished(Merchandise merch) basketFinished(Merchandise merch)} should be called, allowing the basket to be read and updated by registered applications.
25  * Once the basket is finished, {@link #processSale() processSale()} may be called, starting the transaction for current basket total.
26  * As payments are received, the {@link #handleEvent(CommerceEvent event) handleEvent(CommerceEvent event)} method is called with a {@code TRANSACTION_PAYMENT_COMPLETED} event, allowing the appropriate receipt to be printed.
27  * Finally, the {@link #handleEvent(CommerceEvent event) handleEvent(CommerceEvent event)} method is called with a {@code TRANSACTION_ENDED} event, allowing the cashier to begin a new basket and transaction.
28  *
29  * Please note this class is built as an example, and therefore favors simplicity and accessibility over Android best practices.
30  */

63  /**
64  * The listener method. Allows handling of various session events.
65  */
66  public void handleEvent(CommerceEvent event) {
67  switch (event.getType()) {
68  CommerceEvent.SESSION_STARTED:
69  CommerceEvent.SESSION_RESUMED:
70  this.mIsSessionActive = true;
71  break;
72  CommerceEvent.SESSION_ENDED:
73  this.mIsSessionActive = false;
74  break;
75  CommerceEvent.SESSION_CLOSED:
76  // Happens when the session is closed without having endSession() called.
77  CommerceEvent.SESSION_ERROR:
78  CommerceEvent.SESSION_START_FAILED:
79  CommerceEvent.SESSION_RESUME_FAILED:
80  CommerceEvent.SESSION_END_FAILED:
81  // Notify user of error using event.getMessage().
82  this.mIsSessionActive = false;
83  break;
84  TransactionEvent.TRANSACTION_STARTED:
85  /**
86  * Gets the Basket Manager and assigns it to a local variable for use later.
87  * {@code this.mBasketManager = this.mTransactionManager.getBasketManager();}
88  */
89  this.mBasketManager = this.mTransactionManager.getBasketManager();
90  /**
91  * Opens a basket for the transaction, allowing us to start adding merchandise and update the customer-facing display.
92  * {@code this.mBasketManager.openBasket();}
93  */
94  this.mBasketManager.openBasket();
95  break;
96  TransactionEvent.TRANSACTION_PAYMENT_STARTED:
97  // The payment has started. REQUEST events will follow allowing the POS to determine a method of payment, or allow the customer to select on the terminal.
98  break;
99  TransactionEvent.TRANSACTION_PAYMENT_COMPLETED:
100  // Handle the payment completion, printing the receipt in this case.
101  PaymentCompletedEvent payEvent = (PaymentCompletedEvent) event;
102  payEvent.getPayment().getReceipt().print();
103  break;
104  TransactionEvent.TRANSACTION_ENDED:
105  break;
106  TransactionEvent.TRANSACTION_ERROR:
107  // Notify user of error using event.getMessage().
108  break;
109  TransactionEvent.TRANSACTION_FAILED:
110  // Notify user of error using event.getMessage().
111  break;
112  TransactionEvent.TRANSACTION_NOTIFICATION:
113  break;
114  TransactionEvent.TRANSACTION_REQUEST:
115  break;
116  default:
117  break;
118  }
119  return null;
120  }
