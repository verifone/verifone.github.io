75  /**
76  * Sets the adjustments and performs the calculations to change the appropriate amounts.
77  * Please note, this does not change the tax amount, only the total amounts for both the
78  * transaction and the current payment.
79  */
80  public void applyAdjustments(@Nullable final AmountAdjustment[] adjustments) {
81  setAdjustments(adjustments);
82  if (adjustments != null && adjustments.length > 0) {
83  // Separates the transaction and the payment amount to allow multi-tender scenarios.
84  final BigDecimal originalTransactionAmount = getTransaction().getAmount();
85  BigDecimal newTransactionAmount = getTransaction().getAmount();
86  final BigDecimal originalAmount = getPayment().getRequestedPaymentAmount();
87  BigDecimal newRequestedAmount = getPayment().getRequestedPaymentAmount();
88  for (AmountAdjustment adjustment : adjustments) {
89  BigDecimal transactionAugend;
90  BigDecimal paymentAugend;
91  if (adjustment.shouldUsePercentageForAdjustment()) {
92  if (adjustment.getAdjustmentPercentage() != null) {
93  transactionAugend = originalTransactionAmount.multiply(
94  adjustment.getAdjustmentPercentage());
95  paymentAugend = originalAmount.multiply(
96  adjustment.getAdjustmentPercentage());
97  } else {
98  transactionAugend = BigDecimal.ZERO;
99  paymentAugend = BigDecimal.ZERO;
100  }
101  } else {
102  transactionAugend = adjustment.getAdjustmentValue();
103  paymentAugend = adjustment.getAdjustmentValue();
104  }
105  if (transactionAugend != null) {
106  newTransactionAmount = newTransactionAmount.add(transactionAugend);
107  }
108  if (paymentAugend != null) {
109  newRequestedAmount = newRequestedAmount.add(paymentAugend);
110  }
111  }
112  getTransaction().setAmount(newTransactionAmount);
113  getPayment().setRequestedPaymentAmount(newRequestedAmount);
114  }
115  }
