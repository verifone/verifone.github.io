75  /** Assigns a unique basket ID, validating the generation against the current basket. */
76  private void assignBasketItemId(BasketItem item) {
77  if (item.getBasketItemId() == null || item.getBasketItemId().length() == 0) {
78  Date today = Calendar.getInstance().getTime();
79  String basketItemId = ITEM_ID_GENERATOR_FORMAT.format(today);
80  // We probably should optimize this, if the generated ID is the same as the most
81  // recent one, we automatically increase it to make it easier to assign several id's
82  // at once instead of so many list traversals.
83  int loopCounter = 0;
84  while (this.getItemByBasketId(basketItemId) != null && loopCounter < 10) {
85  basketItemId = basketItemId + "" + loopCounter;
86  loopCounter++;
87  }
88  if (loopCounter >= 10) {
89  basketItemId = basketItemId + UUID.randomUUID().toString().substring(0, 5);
90  }
91  item.setBasketItemId(basketItemId);
92  }
93  }

117  /** Helper to traverse lists in the same manner when looking for an item. */
118  private <T extends BasketItem> T getItemFromListByBasketId(String basketItemId,
119  ArrayList<T> list) {
120  if (basketItemId != null && list != null && list.size() > 0) {
121  for (T item : list) {
122  if (item.getBasketItemId().equals(basketItemId))
123  return item;
124  }
125  }
126  return null;

129  /**
130  * Configure's an item's sequence and display order if unset, using the values from the item
131  * if they exceed the current internal state.
132  */
133  private void configureSequenceAndDisplayOrderWithItem(BasketItem item) {
134  // If the item's sequence number is unset, set it to the next number
135  if (item.getSequence() < 0) {
136  this.mSequenceCounter = this.mSequenceCounter + SEQUENCE_INCREMENT;
137  item.setSequence(this.mSequenceCounter);
138  // Else if the item's sequence number is greater than the internal number, use it.
139  } else if (item.getSequence() > this.mSequenceCounter) {
140  this.mSequenceCounter = item.getSequence();
141  }
142  // Else do nothing, it was set elsewhere and should be managed there.
143 
144  // If the item's display order is not set, configure it.
145  if (item.getDisplayOrder() < 0) {
146  this.mDisplayOrder = this.mDisplayOrder + DISPLAY_ORDER_INCREMENT;
147  item.setDisplayOrder(this.mDisplayOrder);
148  // Else if the item's display order is greater than the internal tracking, use it.
149  } else if (item.getDisplayOrder() > this.mDisplayOrder) {
150  this.mDisplayOrder = item.getDisplayOrder();
151  }
152  // Else do nothing, the display order was set elsewhere and should be managed there.
153  }

155  /** Prepares a merchandise and its modifiers for display on screen and ID within the basket. */
156  public void prepareMerchandiseForAdd(Merchandise merchandise) {
157  // Someday, we need to remove the modifications to the display order / sequence from this
158  // stage. That's something that should be managed by the payment processor or the POS,
159  // not by this layer of the service.
160  assignBasketItemId(merchandise);
161  configureSequenceAndDisplayOrderWithItem(merchandise);
162  if (merchandise.getModifierList().size() > 0) {
163  int baseDisplayOrder = merchandise.getDisplayOrder();
164  int displayOrder = baseDisplayOrder;
165  int displayOrderMax = merchandise.getDisplayOrder() + DISPLAY_ORDER_INCREMENT;
166  // Someday, optimize this sequence to not use two loops over the same dataset. Also,
167  // possibly use sorted lists by display order or another good ordering.
168  // First, determine the highest display order to use as the baseline for updating items.
169  for (Modifier modifier : merchandise.getModifierList()) {
170  int comparisonDisplayOrder = modifier.getDisplayOrder();
171  // If the current modifier's display order is not in the range for this
172  // merchandise, adjust it for this merchandise.
173  if (comparisonDisplayOrder > displayOrderMax ||
174  comparisonDisplayOrder < baseDisplayOrder)
175  comparisonDisplayOrder = (comparisonDisplayOrder % DISPLAY_ORDER_INCREMENT) +
176  baseDisplayOrder;
177 
178  // If the modifier's display order is set, use it as a baseline for unset items.
179  if (comparisonDisplayOrder > displayOrder)
180  displayOrder = comparisonDisplayOrder;
181  }
182  // Next, set those items without proper display order.
183  for (Modifier modifier : merchandise.getModifierList()) {
184  int currentDisplayOrder = modifier.getDisplayOrder();
185  // If modifier's display order is unset, give it the next one in the sequence.
186  if (currentDisplayOrder < 0) {
187  displayOrder++;
188  modifier.setDisplayOrder(displayOrder);
189  // Else if the modifier's display order is out of range for the current item, put
190  // it in range appropriately.
191  } else if (currentDisplayOrder > displayOrderMax ||
192  currentDisplayOrder < baseDisplayOrder) {
193  modifier.setDisplayOrder((currentDisplayOrder % DISPLAY_ORDER_INCREMENT) +
194  baseDisplayOrder);
195  }
196  configureSequenceAndDisplayOrderWithItem(modifier);
197  }
198  }

243  public boolean attachModifierToMerchandise(Modifier modifier, Merchandise merchandise) {
244  boolean attachedSuccessfully = false;
245  assignBasketItemId(modifier); // Must happen before attaching to the merchandise.
246  int nextDisplayOrder = merchandise.getDisplayOrder() + 1;
247  for (Modifier preexistingModifier : merchandise.getModifierList()) {
248  if (nextDisplayOrder <= preexistingModifier.getDisplayOrder())
249  nextDisplayOrder = preexistingModifier.getDisplayOrder() + 1;
250  }
251 
252  if (merchandise.attachModifier(modifier)) {
253  modifier.setAttachedToBasketItemId(merchandise.getBasketItemId());
254 
255  attachedSuccessfully = true;
256  }
