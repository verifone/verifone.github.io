587  /**
588  * Merge in the values from the other item, useful when re-synchronizing copied data. Use
589  * aggressively if current valid values should be overriden by the other item.
590  */
591  public void merge(Transaction other, boolean aggressively) {
592  if (other.getParcelVersion() > 1 || TextUtils.isEmpty(other.mType)) {
593  validateAndSetType(other.mTransactionType);
594  } else {
595  validateAndSetType(other.mType);
596  }
597 
598  if (shouldMerge(getOriginator(), other.getOriginator(), aggressively))
599  setOriginator(other.getOriginator());
600  if (shouldMerge(getAcquirerMerchantId(), other.getAcquirerMerchantId(), aggressively))
601  setAcquirerMerchantId(other.getAcquirerMerchantId());
602  if (shouldMerge(getAcquirerTerminalId(), other.getAcquirerTerminalId(), aggressively))
603  setAcquirerTerminalId(other.getAcquirerTerminalId());
604  if (shouldMerge(getAcquirerId(), other.getAcquirerId(), aggressively))
605  setAcquirerId(other.getAcquirerId());
606  if (shouldMerge(getGatewayMerchantId(), other.getGatewayMerchantId(), aggressively))
607  setGatewayMerchantId(other.getGatewayMerchantId());
608  if (shouldMerge(getGatewayTerminalId(), other.getGatewayTerminalId(), aggressively))
609  setGatewayTerminalId(other.getGatewayTerminalId());
610  if (shouldMerge(getInvoiceId(), other.getInvoiceId(), aggressively))
611  setInvoiceId(other.getInvoiceId());
612  if (shouldMerge(getTransactionId(), other.getTransactionId(), aggressively))
613  setTransactionId(other.getTransactionId());
614  if (shouldMerge(getVenue(), other.getVenue(), aggressively))
615  setVenue(other.getVenue());
616  if (shouldMerge(getLane(), other.getLane(), aggressively))
617  setLane(other.getLane());
618  if (shouldMerge(getCashierId(), other.getCashierId(), aggressively))
619  setCashierId(other.getCashierId());
620  if (shouldMerge(getCashierDisplayName(), other.getCashierDisplayName(), aggressively))
621  setCashierDisplayName(other.getCashierDisplayName());
622 
623  if (getAmount().equals(BigDecimal.ZERO) || aggressively)
624  setAmount(other.getAmount());
625  if (shouldMerge(getTaxAmount(), other.getTaxAmount(), aggressively))
626  setTaxAmount(other.getTaxAmount());
627  if (shouldMerge(getGratuityAmount(), other.getGratuityAmount(), aggressively))
628  setGratuityAmount(other.getGratuityAmount());
629  if (shouldMerge(getDonationAmount(), other.getDonationAmount(), aggressively))
630  setDonationAmount(other.getDonationAmount());
631  if (shouldMerge(getCurrency(), other.getCurrency(), aggressively))
632  setCurrency(other.getCurrency());
633 
634  if (!getAllowsManualEntry() || aggressively)
635  setAllowsManualEntry(other.getAllowsManualEntry());
636  if (((getManualPromptOptions() == null || getManualPromptOptions().length == 0)
637  && other.getManualPromptOptions() != null) || aggressively)
638  setManualPromptOptions(other.getManualPromptOptions());
639 
640  if (shouldMerge(getTransactionType(), other.getTransactionType(), aggressively))
641  setTransactionType(other.getTransactionType());
642  if (shouldMerge(getCustomerPhoneNumber(), other.getCustomerPhoneNumber(), aggressively))
643  setCustomerPhoneNumber(other.getCustomerPhoneNumber());
644  if (shouldMerge(getCustomerEmail(), other.getCustomerEmail(), aggressively))
645  setCustomerEmail(other.getCustomerEmail());
646  if (shouldMerge(getResult(), other.getResult(), aggressively))
647  setResult(other.getResult());
648 
649  if (aggressively || (other.getPayments().size() >= getPayments().size()))
650  setPayments(other.getPayments());
651 
652  if (aggressively || other.getBasket() != null)
653  setBasket(other.getBasket());
654  if (aggressively || other.getCurrentPayment() != null)
655  setCurrentPayment(other.getCurrentPayment());
656  if (shouldMerge(getShiftId(), other.getShiftId(), aggressively))
657  setShiftId(other.getShiftId());
658  }
659 
660  private boolean shouldMerge(String current, String other, boolean aggressively) {
661  return (!TextUtils.isEmpty(other) && (TextUtils.isEmpty(current) || aggressively));
662  }
663 
664  private boolean shouldMerge(BigDecimal current, BigDecimal other, boolean aggressively) {
665  return (current == null || aggressively);
666  }
667 
668  private boolean shouldMerge(Integer current, Integer other, boolean aggressively) {
669  return (current == null || aggressively);

691  mTransactionType = in.readString();

696  } else {

702  mAcquirerTerminalId = in.readString();
703  mAcquirerId = in.readString();
