459  /**
460  * Merge in the values from the other item, useful when re-synchronizing copied data. Use
461  * aggressively if current valid values should be overriden by the other item.
462  */
463  public void merge(Transaction other, boolean aggressively) {
464  if (other.getParcelVersion() > 1 || TextUtils.isEmpty(other.mType)) {
465  validateAndSetType(other.mTransactionType);
466  } else {
467  validateAndSetType(other.mType);
468  }
469 
470  if (shouldMerge(getOriginator(), other.getOriginator(), aggressively))
471  setOriginator(other.getOriginator());
472  if (shouldMerge(getAcquirerMerchantId(), other.getAcquirerMerchantId(), aggressively))
473  setAcquirerMerchantId(other.getAcquirerMerchantId());
474  if (shouldMerge(getAcquirerTerminalId(), other.getAcquirerTerminalId(), aggressively))
475  setAcquirerTerminalId(other.getAcquirerTerminalId());
476  if (shouldMerge(getAcquirerId(), other.getAcquirerId(), aggressively))
477  setAcquirerId(other.getAcquirerId());
478  if (shouldMerge(getGatewayMerchantId(), other.getGatewayMerchantId(), aggressively))
479  setGatewayMerchantId(other.getGatewayMerchantId());
480  if (shouldMerge(getGatewayTerminalId(), other.getGatewayTerminalId(), aggressively))
481  setGatewayTerminalId(other.getGatewayTerminalId());
482  if (shouldMerge(getInvoiceId(), other.getInvoiceId(), aggressively))
483  setInvoiceId(other.getInvoiceId());
484  if (shouldMerge(getTransactionId(), other.getTransactionId(), aggressively))
485  setTransactionId(other.getTransactionId());
486  if (shouldMerge(getVenue(), other.getVenue(), aggressively))
487  setVenue(other.getVenue());
488  if (shouldMerge(getLane(), other.getLane(), aggressively))
489  setLane(other.getLane());
490  if (shouldMerge(getCashierId(), other.getCashierId(), aggressively))
491  setCashierId(other.getCashierId());
492  if (shouldMerge(getCashierDisplayName(), other.getCashierDisplayName(), aggressively))
493  setCashierDisplayName(other.getCashierDisplayName());
494 
495  if (getAmount().equals(BigDecimal.ZERO) || aggressively)
496  setAmount(other.getAmount());
497  if (shouldMerge(getTaxAmount(), other.getTaxAmount(), aggressively))
498  setTaxAmount(other.getTaxAmount());
499  if (shouldMerge(getGratuityAmount(), other.getGratuityAmount(), aggressively))
500  setGratuityAmount(other.getGratuityAmount());
501  if (shouldMerge(getDonationAmount(), other.getDonationAmount(), aggressively))
502  setDonationAmount(other.getDonationAmount());
503  if (shouldMerge(getCurrency(), other.getCurrency(), aggressively))
504  setCurrency(other.getCurrency());
505 
506  if (!getAllowsManualEntry() || aggressively)
507  setAllowsManualEntry(other.getAllowsManualEntry());
508  if (((getManualPromptOptions() == null || getManualPromptOptions().length == 0)
509  && other.getManualPromptOptions() != null) || aggressively)
510  setManualPromptOptions(other.getManualPromptOptions());
511 
512  if (shouldMerge(getTransactionType(), other.getTransactionType(), aggressively))
513  setTransactionType(other.getTransactionType());
514  if (shouldMerge(getCustomerPhoneNumber(), other.getCustomerPhoneNumber(), aggressively))
515  setCustomerPhoneNumber(other.getCustomerPhoneNumber());
516  if (shouldMerge(getCustomerEmail(), other.getCustomerEmail(), aggressively))
517  setCustomerEmail(other.getCustomerEmail());
518  if (shouldMerge(getResult(), other.getResult(), aggressively))
519  setResult(other.getResult());
520 
521  if (aggressively || (other.getPayments().size() >= getPayments().size()))
522  setPayments(other.getPayments());
523 
524  if (aggressively || other.getBasket() != null)
525  setBasket(other.getBasket());
526  if (aggressively || other.getCurrentPayment() != null)
527  setCurrentPayment(other.getCurrentPayment());
528  if (shouldMerge(getShiftId(), other.getShiftId(), aggressively))
529  setShiftId(other.getShiftId());
530  }
531

546  public Transaction(Parcel in, int parcelVersion) {
547  super(in, parcelVersion);
548  if (parcelVersion <= 1) {
549  this.mTransactionType = in.readString();
550  }
551  this.mOriginator = in.readString();
552  this.mAcquirerMerchantId = in.readString();
553  this.mAcquirerTerminalId = in.readString();
554  this.mAcquirerId = in.readString();
555  this.mGatewayMerchantId = in.readString();
556  this.mGatewayTerminalId = in.readString();
557  this.mInvoiceId = in.readString();
558  this.mTransactionId = in.readString();
559  this.mVenue = in.readString();
560  this.mLane = in.readString();
561  this.mCashierId = in.readString();
562  this.mCashierDisplayName = in.readString();
563  this.mAmount = ConversionUtility.readBigDecimalFromParcel(in);
564  this.mTaxAmount = ConversionUtility.readBigDecimalFromParcel(in);
565  this.mGratuityAmount = ConversionUtility.readBigDecimalFromParcel(in);
566  this.mDonationAmount = ConversionUtility.readBigDecimalFromParcel(in);
567  this.mCurrency = in.readString();
568  this.mAllowsManualEntry = (in.readInt() == 1);
569  String secondaryTransactionType = in.readString();
570  // Use this one instead of the first if this is available, but don't overwrite the first if it already set and this is empty.
571  if (!TextUtils.isEmpty(secondaryTransactionType)) {
572  this.mTransactionType = secondaryTransactionType;
573  }
574  this.mCustomerPhoneNumber = in.readString();
575  this.mCustomerEmail = in.readString();
576  this.mResult = in.readString();
577  this.mPayments = in.createTypedArrayList(Payment.CREATOR);
578  String[] stringPromptOptions = in.createStringArray();
579  if (stringPromptOptions != null) {
580  ManualPromptOption[] promptOptions = new ManualPromptOption[stringPromptOptions.length];
581  for (int i = 0; i < stringPromptOptions.length; i++) {
582  promptOptions[i] = ManualPromptOption.valueOf(stringPromptOptions[i]);
583  }
584  this.mManualPromptOptions = promptOptions;
585  }
586  this.mBasket = in.readParcelable(Basket.class.getClassLoader());
587  this.mCurrentPayment = in.readParcelable(Payment.class.getClassLoader());
588  if (in.dataAvail() > 0) {
589  this.mShiftId = ConversionUtility.readIntegerFromParcel(in);
590  }
591  }

593  @Override
594  public void setParcelVersion(int targetParcelVersion) {
595  super.setParcelVersion(targetParcelVersion);
596  if (getPayments() != null) {
597  for (Payment payment : getPayments()) {
598  payment.setParcelVersion(targetParcelVersion);
599  }
600  }
601  if (getCurrentPayment() != null) {
602  getCurrentPayment().setParcelVersion(targetParcelVersion);
603  }
604  if (getBasket() != null) {
605  getBasket().setParcelVersion(targetParcelVersion);
606  }
607  // Just in case someone accidentally sets the reference transaction to itself, we
608  // won't loop infinitely.
609  if (getReferenceTransaction() != null && getReferenceTransaction() != this) {
610  getReferenceTransaction().setParcelVersion(targetParcelVersion);
611  }
612  }
