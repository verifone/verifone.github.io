225  /**
226  * Merge in the values from the other item, useful when re-synchronizing copied data. Use
227  * aggressively if current valid values should be overridden by the other item.
228  */
229  @CallSuper
230  public void merge(BasketItem item, boolean aggressively) {
231  if (getBasketItemId().equals(item.getBasketItemId())) {
232  // We have to check if the other item is set to the default before setting, because
233  // we set this value internally and don't want to accidentally remove it. Someday,
234  // we'll remove this requirement once we fix the basket wrapper to not update this
235  // value.
236  if (item.getSequence() != UNSET_VALUE
237  && (aggressively || getSequence() == UNSET_VALUE))
238  setSequence(item.getSequence());
239  // We have to check if the other item is set to the default before setting, because
240  // we set this value internally and don't want to accidentally remove it. Someday,
241  // we'll remove this requirement once we fix the basket wrapper to not update this
242  // value.
243  if (item.getDisplayOrder() != UNSET_VALUE
244  && (aggressively || getDisplayOrder() == UNSET_VALUE))
245  setDisplayOrder(item.getDisplayOrder());
246  if (aggressively || TextUtils.isEmpty(getDisplayLine()))
247  setDisplayLine(item.getDisplayLine());
248  if (aggressively || TextUtils.isEmpty(getName()))
249  setName(item.getName());
250  if (aggressively || TextUtils.isEmpty(getDescription()))
251  setDescription(item.getDescription());
252  if (aggressively || TextUtils.isEmpty(getSku()))
253  setSku(item.getSku());
254  if (aggressively || TextUtils.isEmpty(getUpc()))
255  setUpc(item.getUpc());
256  if (item.getRemoved() || aggressively)
257  setRemoved(item.getRemoved());
258  if (aggressively || getTax() == null)
259  setTax(item.getTax());
260  if (aggressively || getAmount() == null)
261  setAmount(item.getAmount());
262  if (aggressively || getAmountTotals() == null
263  || (getAmountTotals().isUnset()
264  && item.getAmountTotals() != null))
265  setAmountTotals(item.getAmountTotals());
266  }
267  }

313  public BasketItem(Parcel in, int parcelVersion) {
314  super(in, parcelVersion);
315  if (parcelVersion < 12 && getClass().getName().equals(Donation.class.getName())) {
316  // We changed the class inheritance for Donation in version 12 to extend this class, so
317  // if it's an earlier version, we need to skip the below reads.
318  return;
319  }
320  setBasketItemId(in.readString());
321  setSequence(in.readInt());
322  if (getParcelVersion() >= 0) {
323  setDisplayOrder(in.readInt());
324  setDisplayLine(in.readString());
325  setName(in.readString());
326  setDescription(in.readString());
327  setSku(in.readString());
328  setUpc(in.readString());
329  setAmount(ConversionUtility.readBigDecimalFromParcel(in));
330  setTax(ConversionUtility.readBigDecimalFromParcel(in));
331  setRemoved((in.readInt() == 1));
332  mAmountTotals = in.readParcelable(AmountTotals.class.getClassLoader());
333  }
