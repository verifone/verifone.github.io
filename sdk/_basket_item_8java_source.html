181  /**
182  * Merge in the values from the other item, useful when re-synchronizing copied data. Use
183  * aggressively if current valid values should be overridden by the other item.
184  */
185  public void merge(BasketItem item, boolean aggressively) {
186  if (this.getBasketItemId().equals(item.getBasketItemId())) {
187  // We have to check if the other item is set to the default before setting, because
188  // we set this value internally and don't want to accidentally remove it. Someday,
189  // we'll remove this requirement once we fix the basket wrapper to not update this
190  // value.
191  if (item.getSequence() != UNSET_VALUE
192  && (aggressively || this.getSequence() == UNSET_VALUE))
193  this.setSequence(item.getSequence());
194  // We have to check if the other item is set to the default before setting, because
195  // we set this value internally and don't want to accidentally remove it. Someday,
196  // we'll remove this requirement once we fix the basket wrapper to not update this
197  // value.
198  if (item.getDisplayOrder() != UNSET_VALUE
199  && (aggressively || this.getDisplayOrder() == UNSET_VALUE))
200  this.setDisplayOrder(item.getDisplayOrder());
201  if (aggressively || TextUtils.isEmpty(this.getDisplayLine()))
202  this.setDisplayLine(item.getDisplayLine());
203  if (aggressively || TextUtils.isEmpty(this.getName()))
204  this.setName(item.getName());
205  if (aggressively || TextUtils.isEmpty(this.getDescription()))
206  this.setDescription(item.getDescription());
207  if (aggressively || TextUtils.isEmpty(this.getSku()))
208  this.setSku(item.getSku());
209  if (aggressively || TextUtils.isEmpty(this.getUpc()))
210  this.setUpc(item.getUpc());
211  if (item.getRemoved() || aggressively)
212  this.setRemoved(item.getRemoved());
213  if (aggressively || this.getTax() == null)
214  this.setTax(item.getTax());
215  if (aggressively || this.getAmount() == null)
216  this.setAmount(item.getAmount());
217  if (aggressively || this.getAmountTotals() == null
218  || (this.getAmountTotals().isUnset()
219  && item.getAmountTotals() != null))
220  this.setAmountTotals(item.getAmountTotals());
221  }
222  }

224  /**
225  * Sorts basket items by {@link #getBasketItemId()}, allowing faster comparison and other
226  * operations when managing multiple items. Currently returns equal if
227  * {@link #getBasketItemId()} returns null for both items.
228  */
229  @Override
230  public int compareTo(BasketItem other) {
231  if (this.getBasketItemId() != null && other.getBasketItemId() != null) {
232  return this.getBasketItemId().compareTo(other.getBasketItemId());
233  } else if (this.getBasketItemId() != null) {
234  return 1;
235  } else if (other.getBasketItemId() != null) {
236  return -1;
237  } else {
238  return 0;
239  }
240  }
