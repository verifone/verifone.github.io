197  @Override
198  public void merge(BasketItem item, boolean aggressively) {
199  if (this.getBasketItemId().equals(item.getBasketItemId())
200  && Merchandise.class.isInstance(item)) {
201  super.merge(item, aggressively);
202  Merchandise other = (Merchandise) item;
203  if (aggressively || this.getUnitPrice() == null)
204  this.setUnitPrice(other.getUnitPrice());
205  if (aggressively || this.getQuantity() == null || BigDecimal.ONE.compareTo(getQuantity()) == 0)
206  this.setQuantity(other.getQuantity());
207  if (aggressively || this.getUnitOfMeasurement() == null) {
208  this.setUnitOfMeasurement(other.getUnitOfMeasurement());
209  }
210  if (aggressively || this.getExtendedPrice() == null)
211  this.setExtendedPrice(other.getExtendedPrice());
212  if (aggressively || this.getDiscount() == null)
213  this.setDiscount(other.getDiscount());
214 
215  // Need to merge modifiers. This is a bit trickier.
216  ArrayList<Modifier> otherMods = null;
217  if (other.mModifiers != null)
218  otherMods = new ArrayList<>(other.mModifiers);
219  boolean otherHasMods = otherMods != null && otherMods.size() > 0;
220  boolean thisHasMods = mModifiers != null && mModifiers.size() > 0;
221  if (otherHasMods && !thisHasMods) {
222  Log.v(TAG, "Merging modifiers: otherHasMods && !thisHasMods");
223  mModifiers = otherMods;
224  } else if (otherHasMods) { // This has mods implicitly at this point.
225  ArrayList<Modifier> iterableMods = new ArrayList<>(mModifiers);
226  Collections.sort(iterableMods);
227  Collections.sort(otherMods);
228  Log.v(TAG, "Merging modifiers: otherHasMods\n" + iterableMods + "\n" + otherMods);
229  int curPos = 0;
230  final int otherSize = otherMods.size();
231  for (Modifier modifier : iterableMods) {
232  Modifier matchingMod = null;
233  boolean shouldKeepSearching = true;
234  while (curPos < otherSize && matchingMod == null && shouldKeepSearching) {
235  Modifier otherMod = otherMods.get(curPos);
236  int comparison = modifier.compareTo(otherMod);
237  if (comparison < 0) {
238  shouldKeepSearching = false;
239  } else {
240  if (comparison == 0)
241  matchingMod = otherMod;
242  curPos++;
243  }
244  }
245  Log.v(TAG, "Matching mod: " + matchingMod);
246  if (matchingMod != null)
247  modifier.merge(matchingMod, aggressively);
248  else if (aggressively)
249  mModifiers.remove(modifier); // This could be optimized.
250  // else leave the current modifier untouched.
251  }
252  } else if (aggressively) {
253  mModifiers = otherMods;
254  }
255  }
256  }

261  public Merchandise(Parcel in, int parcelVersion) {
262  super(in, parcelVersion);
263  if (parcelVersion < 0) {
264  setName(in.readString());
265  setDescription(in.readString());
266  setDisplayLine(in.readString());
267  setQuantity(in.readInt());
268  setSku(in.readString());
269  setUpc(in.readString());
270  setRemoved(in.readByte() != 0);
271  operation = in.readInt();
272  } else {
273  this.mUnitPrice = ConversionUtility.readBigDecimalFromParcel(in);
274  if (getParcelVersion() <= 3) {
275  this.mQuantity = new BigDecimal(in.readInt());
276  }
277  this.mExtendedPrice = ConversionUtility.readBigDecimalFromParcel(in);
278  this.mDiscount = ConversionUtility.readBigDecimalFromParcel(in);
279  this.operation = in.readInt();
280  in.readTypedList(this.mModifiers, Modifier.CREATOR);
281  if (getParcelVersion() > 3) {
282  this.mQuantity = ConversionUtility.readBigDecimalFromParcel(in);
283  this.mUnitOfMeasurement = in.readString();
284  }
285  }

346  @SuppressWarnings("unchecked")
347  @Override
348  protected <CPEntityType extends CPBaseParcel> CPEntityType buildFromCpJson(@NonNull JSONObject jsonObject, @Nullable CPEntityType parentObject) {
349  Merchandise targetObject = this;
350  if (parentObject != null) {
351  targetObject = (Merchandise) parentObject;
352  }
353  targetObject = super.buildFromCpJson(jsonObject, targetObject);
354 
355  final String defaultFallback = null;
356  String value;
357  value = jsonObject.optString("Unit_Price", defaultFallback);
358  if (value != null) {
359  targetObject.setUnitPrice(ConversionUtility.parseAmount(value));
360  }
361 
362  value = jsonObject.optString("Extended_Price", defaultFallback);
363  if (value != null) {
364  targetObject.setExtendedPrice(ConversionUtility.parseAmount(value));
365  }
366 
367  value = jsonObject.optString("Quantity", defaultFallback);
368  if (value != null) {
369  try {
370  targetObject.setQuantity(Integer.parseInt(value));
371  } catch (NumberFormatException e) {
372  Log.w(TAG, "Quantity can not be parsed: " + e.getMessage());
373  }
374  }
375 
376  return (CPEntityType) targetObject;
377  }
