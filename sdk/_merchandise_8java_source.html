155  @Override
156  public void merge(BasketItem item, boolean aggressively) {
157  if (this.getBasketItemId().equals(item.getBasketItemId())
158  && Merchandise.class.isInstance(item)) {
159  super.merge(item, aggressively);
160  Merchandise other = (Merchandise) item;
161  if (aggressively || this.getUnitPrice() == null)
162  this.setUnitPrice(other.getUnitPrice());
163  if (aggressively || this.getQuantity() == DEFAULT_QUANTITY)
164  this.setQuantity(other.getQuantity());
165  if (aggressively || this.getExtendedPrice() == null)
166  this.setExtendedPrice(other.getExtendedPrice());
167  if (aggressively || this.getDiscount() == null)
168  this.setDiscount(other.getDiscount());
169 
170  // Need to merge modifiers. This is a bit trickier.
171  ArrayList<Modifier> otherMods = null;
172  if (other.mModifiers != null)
173  otherMods = new ArrayList<>(other.mModifiers);
174  boolean otherHasMods = otherMods != null && otherMods.size() > 0;
175  boolean thisHasMods = mModifiers != null && mModifiers.size() > 0;
176  if (otherHasMods && !thisHasMods) {
177  Log.v(TAG, "Merging modifiers: otherHasMods && !thisHasMods");
178  mModifiers = otherMods;
179  } else if (otherHasMods) { // This has mods implicitly at this point.
180  ArrayList<Modifier> iterableMods = new ArrayList<>(mModifiers);
181  Collections.sort(iterableMods);
182  Collections.sort(otherMods);
183  Log.v(TAG, "Merging modifiers: otherHasMods\n" + iterableMods + "\n" + otherMods);
184  int curPos = 0;
185  final int otherSize = otherMods.size();
186  for (Modifier modifier : iterableMods) {
187  Modifier matchingMod = null;
188  boolean shouldKeepSearching = true;
189  while (curPos < otherSize && matchingMod == null && shouldKeepSearching) {
190  Modifier otherMod = otherMods.get(curPos);
191  int comparison = modifier.compareTo(otherMod);
192  if (comparison < 0) {
193  shouldKeepSearching = false;
194  } else {
195  if (comparison == 0)
196  matchingMod = otherMod;
197  curPos++;
198  }
199  }
200  Log.v(TAG, "Matching mod: " + matchingMod);
201  if (matchingMod != null)
202  modifier.merge(matchingMod, aggressively);
203  else if (aggressively)
204  mModifiers.remove(modifier); // This could be optimized.
205  // else leave the current modifier untouched.
206  }
207  } else if (aggressively) {
208  mModifiers = otherMods;
209  }
210  }
