69  if (this.getBasketItemId().equals(item.getBasketItemId())) {
70  // We have to do this check here for the amount, since the donation amount defaults to
71  // 0, but a normal basket item amount defaults to null.
72  if (!item.getAmount().equals(ConversionUtility.DEFAULT_BIG_DECIMAL)
73  && (aggressively || getAmount().equals(ConversionUtility.DEFAULT_BIG_DECIMAL))) {
74  setAmount(item.getAmount());
75  }
