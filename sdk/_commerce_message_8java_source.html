104  public static CommerceMessage buildMessageFromIntent(@NonNull final Intent intent) {
105  CommerceMessage message = null;
106  final String action = intent.getAction();
107  if (action != null && action.length() > INTENT_ACTION_PREFIX.length()) {
108  final String messageName = action.substring(INTENT_ACTION_PREFIX.length());
109  final String messageClassName = getMessageClassName(messageName);
110 
111  if (messageClassName != null) {
112  message = loadMessageByName(messageClassName);
113  if (message != null) {
114  populateMessageFromIntent(message, intent);
115  }
116  }
117  }
118  return message;
119  }

162  @Nullable
163  private static String getMessageClassName(@NonNull final String messageName) {
164  String nameToSearch = null;
165  if (messageName.startsWith(CommerceAppMessage.NAME)) {
166  // Commerce App Message.
167  return "CommerceAppMessage";
168  } else if (messageName.startsWith(CommerceApi.MESSAGE_NAME_PREFIX)) {
169  nameToSearch = messageName.substring(CommerceApi.MESSAGE_NAME_PREFIX.length());
170  } else if (messageName.startsWith(CommerceTrigger.MESSAGE_NAME_PREFIX)) {
171  nameToSearch = messageName.substring(CommerceTrigger.MESSAGE_NAME_PREFIX.length());
172  }
173  if (nameToSearch == null) {
174  return null;
175  }
176  StringBuilder sourceClassName = new StringBuilder();
177  Matcher matcher = NAME_TRANSFORMER_PATTERN.matcher(nameToSearch);
178  while (matcher.find()) {
179  if (matcher.groupCount() >= 2) {
180  sourceClassName.append(matcher.group(1));
181  sourceClassName.append(matcher.group(2).toLowerCase());
182  }
183  }
184  boolean addedPostfix =
185  (TYPE_CONVERSION.NOTIFIES.addPostfixIfMatching(nameToSearch, sourceClassName)
186  || TYPE_CONVERSION.REQUESTS.addPostfixIfMatching(nameToSearch, sourceClassName)
187  || TYPE_CONVERSION.RECEIVES.addPostfixIfMatching(nameToSearch, sourceClassName));
188  if (!addedPostfix) {
189  Log.w(TAG, "Unrecognized prefix for " + nameToSearch + ".");
190  }
191  return CommerceTrigger.class.getPackage().getName() + "." + sourceClassName.toString();
192  }
