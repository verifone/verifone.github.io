170  private static Intent createExplicitFromImplicitIntent(Context context, Intent implicitIntent) {
171  //Retrieve all services that can match the given intent
172  PackageManager pm = context.getPackageManager();
173  List<ResolveInfo> resolveInfo = pm.queryIntentServices(implicitIntent, 0);
174 
175  //Make sure only one match was found
176  if (resolveInfo == null || resolveInfo.size() != 1) {
177  Log.e(TAG, "createExplicitFromImplicitIntent Query intent service failed.");
178  return null;
179  }
180 
181  //Get component info and create ComponentName
182  ResolveInfo serviceInfo = resolveInfo.get(0);
183  String packageName = serviceInfo.serviceInfo.packageName;
184  String className = serviceInfo.serviceInfo.name;
185  ComponentName component = new ComponentName(packageName, className);
186 
187  //Create a new intent. Use the old one for extras and such reuse
188  Intent explicitIntent = new Intent(implicitIntent);
189 
190  //Set the component to be explicit
191  explicitIntent.setComponent(component);
192 
193  return explicitIntent;
194  }
