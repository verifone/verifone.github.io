22 /**
23  * The parent class for the events that may occur during a transaction. This is only ever
24  * generated by the underlying service, and passed to the appropriate listener.
25  */
26 public class TransactionEvent extends CommerceEvent {
27  /** A general transaction event notifying listeners that something has happened. */
28  public static final String TRANSACTION_NOTIFICATION = "TRANSACTION_NOTIFICATION";
29  /** A general transaction event requesting information from listeners. */
30  public static final String TRANSACTION_REQUEST = "TRANSACTION_REQUEST";
31  /** The event type when a payment has started. */
32  public static final String TRANSACTION_PAYMENT_STARTED = "TRANSACTION_PAYMENT_STARTED";
33  /** The event type when a payment has been completed. */
34  public static final String TRANSACTION_PAYMENT_COMPLETED = "TRANSACTION_PAYMENT_COMPLETED";
35  /** The event type when an action has started for non-Payment types. */
36  public static final String TRANSACTION_STARTED = "TRANSACTION_STARTED";
37  /** The event type when an action has ended for non-Payment types. */
38  public static final String TRANSACTION_ENDED = "TRANSACTION_ENDED";
39  /** A general type when an error has occurred during the transaction. */
40  public static final String TRANSACTION_ERROR = "TRANSACTION_ERROR";
41  /** The transaction has failed. It will need to be started again. */
42  public static final String TRANSACTION_FAILED = "TRANSACTION_FAILED";
43  /** The login process is complete, check the status for the success/failure of the attempt. */
44  public static final String LOGIN_COMPLETED = "LOGIN_COMPLETED";
45  /** The logout process is complete, check the status for the success/failure of the attempt. */
46  public static final String LOGOUT_COMPLETED = "LOGOUT_COMPLETED";

67  @Override

75  @Override
76  public Response generateResponse() {
77  return new Response(this.getSessionId(), null, this.getInvoiceId(), this.getTransaction());
78  }

80  /* For making the code simpler in the various managers. */
81  static TransactionEvent getGeneralErrorEvent(String sessionId, String message) {
82  return new TransactionEvent(sessionId, -1, Status.STATUS_ERROR, message);
83  }

85  /* For making the code simpler in the various managers. */

123  /**
124  * The parent class for responses to transaction events.
125  * This and the subclasses should only be generated by the generateResponse() function
126  * of the matching event.
127  */
128  public static class Response extends CommerceEvent.Response {
129  private String mInvoiceId;
130  private Transaction mTransaction;
131 
132  /** The invoice ID for the transaction. */
133  public String getInvoiceId() { return this.mInvoiceId; }
134 
135  /**
136  * The transaction object. Making changes to the transaction object does not affect the
137  * actual object used by the underlying service.
138  */
139  public Transaction getTransaction() { return this.mTransaction; }
140 
141  protected Response (String sessionId, String type,
142  String invoiceId, Transaction transaction) {
143  super(sessionId, type);
144  this.mInvoiceId = invoiceId;
145  this.mTransaction = transaction;
146  }
147 
148  public Response(Parcel in, int parcelVersion) {
149  super(in, parcelVersion);
150  this.mInvoiceId = in.readString();
151  this.mTransaction = in.readParcelable(Transaction.class.getClassLoader());
152  }
153 
154  @Override
155  public void setParcelVersion(int targetParcelVersion) {
156  super.setParcelVersion(targetParcelVersion);
157  getTransaction().setParcelVersion(targetParcelVersion);
158  }
159 
160  @Override
161  public void writeToParcel(Parcel dest, int flags) {
162  super.writeToParcel(dest, flags);
163  dest.writeString(this.mInvoiceId);
164  dest.writeParcelable(this.mTransaction, flags);
165  }
166 
167  /** Parcelable creator, returning a concrete implementation. */
168  @SuppressWarnings ("unchecked")
169  public static final Parcelable.Creator<Response> CREATOR =
170  new Response.ParcelCreator<Response>() {
171  // We override at this point, because we have to guarantee that the
172  // transaction is present.
173  @Override
174  public Response createFromParcel(Parcel source) {
175  CommerceResponse createdObject = super.createFromParcel(source);
176  if (createdObject == null ||
177  !CommerceResponse.class.isInstance(createdObject))
178  return new Response(source, SDK_PARCEL_VERSION);
179  else
180  return (Response) createdObject;
181  }
182  // We override at this point, because it optimizes the result, as this
183  // version of the response, containing the transaction object, is
184  // significantly larger than the super classes.
185  @Override
186  public Response[] newArray(int size) {
187  return new Response[size];
188  }
189  };
190  }
