20 /**
21  * The parent class for the events that may occur during a transaction. This is only ever
22  * generated by the underlying service, and passed to the appropriate listener.
23  */
24 public class TransactionEvent extends CommerceEvent {
25  /** A general transaction event notifying listeners that something has happened. */
26  public static final String TRANSACTION_NOTIFICATION = "TRANSACTION_NOTIFICATION";
27  /** A general transaction event requesting information from listeners. */
28  public static final String TRANSACTION_REQUEST = "TRANSACTION_REQUEST";
29  /** The event type when a payment has started. */
30  public static final String TRANSACTION_PAYMENT_STARTED = "TRANSACTION_PAYMENT_STARTED";
31  /**
32  * The event type when a payment requires online authorization. An event with this type will
33  * only be sent when the client is performing online authorization itself.
34  */
35  public static final String TRANSACTION_PAYMENT_REQUIRES_ONLINE_AUTHORIZATION = "TRANSACTION_PAYMENT_REQUIRES_ONLINE_AUTHORIZATION";
36  /** The event type when a payment has been completed. */
37  public static final String TRANSACTION_PAYMENT_COMPLETED = "TRANSACTION_PAYMENT_COMPLETED";
38  /** The event type when an action has started for non-Payment types. */
39  public static final String TRANSACTION_STARTED = "TRANSACTION_STARTED";
40  /** The event type when an action has ended for non-Payment types. */
41  public static final String TRANSACTION_ENDED = "TRANSACTION_ENDED";
42  /** A general type when an error has occurred during the transaction. */
43  public static final String TRANSACTION_ERROR = "TRANSACTION_ERROR";
44  /** The transaction has failed. It will need to be started again. */
45  public static final String TRANSACTION_FAILED = "TRANSACTION_FAILED";
46  /** The login process is complete, check the status for the success/failure of the attempt. */
47  public static final String LOGIN_COMPLETED = "LOGIN_COMPLETED";
48  /** The logout process is complete, check the status for the success/failure of the attempt. */
49  public static final String LOGOUT_COMPLETED = "LOGOUT_COMPLETED";

70  /**
71  * The generator method for generating the appropriate response for the current event.
72  * @return empty response when a standard transaction event. Child classes may implement
73  * appropriate responses as needed.
74  */
75  @Override
76  public Response generateResponse() {
77  if (TransactionEvent.TRANSACTION_PAYMENT_REQUIRES_ONLINE_AUTHORIZATION.equals(getType())) {
78  return new OnlineAuthorizationResponse(getSessionId(), getType(), getInvoiceId(),
79  getTransaction());
80  } else {
81  return new Response(getSessionId(), null, getInvoiceId(), getTransaction());
82  }
83  }

209  /** Parcelable creator, returning a concrete implementation. */
210  @SuppressWarnings ("unchecked")
211  public static final Parcelable.Creator<Response> CREATOR =
212  new Response.ParcelCreator<Response>() {
213  // We override at this point, because we have to guarantee that the
214  // transaction is present.
215  @Override
216  public Response createFromParcel(Parcel source) {
217  CommerceResponse createdObject = super.createFromParcel(source);
218  if (createdObject == null || !CommerceResponse.class.isInstance(createdObject)) {
219  return new Response(source, getRecommendedParcelVersion());
220  } else {
221  return (Response) createdObject;
222  }
223  }
224  // We override at this point, because it optimizes the result, as this
225  // version of the response, containing the transaction object, is
226  // significantly larger than the super classes.
227  @Override
228  public Response[] newArray(int size) {
229  return new Response[size];
230  }
231  };
232  }
