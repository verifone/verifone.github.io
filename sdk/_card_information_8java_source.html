611  @SuppressWarnings("unchecked")
612  @Override
613  protected <CPEntityType extends CPBaseParcel> CPEntityType buildFromCpJson(@NonNull JSONObject jsonObject, @Nullable CPEntityType parentObject) {
614  CardInformation targetObject = this;
615  if (parentObject != null) {
616  targetObject = (CardInformation) parentObject;
617  }
618 
619  // Build card information object.
620  final String defaultFallback = null;
621  final JSONArray preferredLanguages = jsonObject.optJSONArray("Card_Preferred_Languages");
622  final String panHandle = jsonObject.optString("PAN_Handle", defaultFallback);
623  final String panLast4 = jsonObject.optString("PAN_Last4", defaultFallback);
624  final String cardPan = jsonObject.optString("Card_PAN", defaultFallback);
625  final String cardCurrency = jsonObject.optString("Card_Currency", defaultFallback);
626  final String cardCountry = jsonObject.optString("Card_Country", defaultFallback);
627 
628  if (preferredLanguages != null || panHandle != null || panLast4 != null || cardPan != null || cardCurrency != null || cardCountry != null) {
629  if (panHandle != null) {
630  targetObject.setPanHandle(panHandle);
631  }
632  if (panLast4 != null) {
633  targetObject.setPanLast4(panLast4);
634  }
635  if (cardPan != null) {
636  targetObject.setCardPan(cardPan);
637  }
638  if (cardCurrency != null) {
639  targetObject.setCardCurrency(cardCurrency);
640  }
641  if (cardCountry != null) {
642  targetObject.setCardCountry(cardCountry);
643  }
644  if (preferredLanguages != null && preferredLanguages.length() > 0) {
645  try {
646  int length = preferredLanguages.length();
647  String[] values = new String[length];
648  for (int i = 0; i < length; i++) {
649  values[i] = preferredLanguages.getString(i);
650  }
651  targetObject.setCardPreferredLanguages(TextUtils.join(LANGUAGES_SEPARATOR, values));
652  } catch (JSONException e) {
653  Log.w(TAG, "Exception parsing preferred languages. " + e.getMessage());
654  }
655  }
656  final String presentationMethodString = jsonObject.optString("Card_Type", defaultFallback);
657  if (presentationMethodString != null) {
658  if ("CTLS".equals(presentationMethodString)) {
659  targetObject.setPresentationMethod(CardInformation.PresentationMethod.CTLS_CARD);
660  } else {
661  for (CardInformation.PresentationMethod method : CardInformation.PresentationMethod.values()) {
662  if (method.name().equals(presentationMethodString)) {
663  targetObject.setPresentationMethod(method);
664  }
665  }
666  }
667  }
668  }
669 
670  final String bin = jsonObject.optString("BIN", defaultFallback);
671  if (bin != null) {
672  targetObject.setBin(bin);
673  }
674 
675  final String cardHolderName = jsonObject.optString("Card_Holder_Name", defaultFallback);
676  if (cardHolderName != null) {
677  targetObject.setCardHolderName(cardHolderName);
678  }
679 
680  final String cardTrack1 = jsonObject.optString("Card_Track1", defaultFallback);
681  if (cardTrack1 != null) {
682  targetObject.setCardTrack1(cardTrack1);
683  }
684 
685  final String cardTrack2 = jsonObject.optString("Card_Track2", defaultFallback);
686  if (cardTrack2 != null) {
687  targetObject.setCardTrack2(cardTrack2);
688  }
689 
690  final String cardTrack3 = jsonObject.optString("Card_Track3", defaultFallback);
691  if (cardTrack3 != null) {
692  targetObject.setCardTrack3(cardTrack3);
693  }
694 
695  final String cardExpiry = jsonObject.optString("Card_Expiry", defaultFallback);
696  if (cardExpiry != null) {
697  targetObject.setCardExpiry(cardExpiry);
698  }
699 
700  final String cardStatus = jsonObject.optString("Card_Status", defaultFallback);
701  if (cardStatus != null) {
702  targetObject.setCardStatus(cardStatus);
703  }
704 
705  final String encryptedPAN = jsonObject.optString("Encrypted_PAN", defaultFallback);
706  if (encryptedPAN != null) {
707  targetObject.setEncryptedPan(encryptedPAN);
708  }
709 
710  final String encryptedPANKSN = jsonObject.optString("Encrypted_PAN_KSN", defaultFallback);
711  if (encryptedPANKSN != null) {
712  targetObject.setEncryptedPanKsn(encryptedPANKSN);
713  }
714 
715  return (CPEntityType) targetObject;
716  }

718  public void writeToJson(JSONObject jsonObject) throws JSONException {
719  String preferredLanguages = getCardPreferredLanguages();
720  if (preferredLanguages != null) {
721  String[] languages = TextUtils.split(preferredLanguages, LANGUAGES_SEPARATOR);
722  JSONArray languageArray = new JSONArray(languages);
723  jsonObject.put("Card_Preferred_Languages", languageArray);
724  }
725 
726  jsonObject.putOpt("PAN_Handle", getPanHandle());
727  jsonObject.putOpt("PAN_Last4", getPanLast4());
728  jsonObject.putOpt("Card_Currency", getCardCurrency());
729  jsonObject.putOpt("Card_Country", getCardCountry());
730 
731  PresentationMethod method = getPresentationMethod();
732  if (method == PresentationMethod.CTLS_CARD) {
733  jsonObject.putOpt("Card_Type", "CTLS");
734  } else if (method != null) {
735  jsonObject.putOpt("Card_Type", method.name());
736  }
737 
738  jsonObject.putOpt("BIN", getBin());
739  }
