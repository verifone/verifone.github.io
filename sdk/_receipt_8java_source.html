583  @NonNull
584  static String getHtmlStringOrEmptyComment(@Nullable String divId,
585  @Nullable String divClass,
586  @Nullable String htmlString) {
587  if (htmlString != null) {
588  boolean shouldWrapInDiv = (!TextUtils.isEmpty(divId) || !TextUtils.isEmpty(divClass));
589  if (shouldWrapInDiv) {
590  String divWrapper = "<div";
591  if (!TextUtils.isEmpty(divId))
592  divWrapper = divWrapper + " id=\"" + divId + "\"";
593  if (!TextUtils.isEmpty(divClass))
594  divWrapper = divWrapper + " class=\"" + divClass + "\"";
595  // If the html string already contains the div wrapper, then we've already
596  // formatted it, and shouldn't wrap it again. This is particularly possible with
597  // the Transaction Information section, since it needs to be included with the
598  // original HTML as well.
599  if (!htmlString.startsWith(divWrapper))
600  return divWrapper + ">" + htmlString + "</div>";
601  else
602  return htmlString;
603  } else {
604  return htmlString;
605  }
606  } else {
607  return "<!-- Empty -->";
608  }
609  }
