33  switch (status) {

40  Peripheral.STATUS_READY:

42  break;
43  Peripheral.STATUS_BUSY:
44  /** For {@code Peripheral.STATUS_BUSY}, possibly display an error to the user. This will not happen with a cash drawer, as it cannot be reserved. */
45  break;
46  Peripheral.STATUS_DISCONNECTED:
47  /** For {@code Peripheral.STATUS_DISCONNECTED}, possibly display an error to the user. */
48  break;
49  Peripheral.STATUS_ERROR:
50  /** For {@code Peripheral.STATUS_ERROR}, possibly display an error to the user. */
51  break;
52  Peripheral.STATUS_UNKNOWN:
53  /** For {@code Peripheral.STATUS_UNKNOWN}, this should probably be ignored, as future statuses without a proper adaptation to this version of the API will be labeled as this. */
54  break;
55  default:
56  break;
57  }
