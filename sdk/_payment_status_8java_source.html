14 package com.verifone.commerce.payment;
15 
16 import android.os.Parcel;
17 import android.os.Parcelable;
18 
19 import lombok.Data;
20 
21 @Data
22 public class PaymentStatus implements Parcelable{
23 
24  /**
25  * @transId
26  * Transaction Id as provided by the payment gateway. It is a client-specific transaction identifier. This transId is used for subsequent payment transactions such as VOID. This is a mandatory field.
27  */
28  private String transId;
29 
30  /**
31  * @holderAuth
32  * Means by which the transaction is authorized. Must be one of: PIN, SIGNATURE, ID, OTHER.
33  */
34  private String holderAuth;
35 
36 
37  /**
38  * @invoice
39  * Unique reference to transaction. Ex: 72135
40  */
41  private String invoice;
42 
43  /**
44  * @storeNum
45  * Store Number. Uniquely identifies a store in a store chain. Ex: 1091
46  */
47  private String storeNum;
48 
49  /**
50  * @lane
51  * Lane the payment terminal is located in the store. Ex: 9
52  */
53  private String lane;
54 
55  /**
56  * @cashierId
57  * Reference to the Cashier. Ex: Cathy
58  */
59  private String cashierId;
60 
61  /**
62  * @transactionType
63  * Type of transaction executed by the Payment Application and must one of: PAYMENT, DEFERRED_PAYMENT, DEFERRED_PAYMENT_COMPLETION, REFUND, VOID, PRE_AUTHORISATION, PRE_AUTHORISATION_COMPLETION, CASH_ADVANCE, CASHBACK
64  */
65  private String transactionType;
66 
67  /**
68  * @transAmount
69  * Transaction Amount. String amount formatting applies. 18.90
70  */
71  private String transAmount;
72 
73  /**
74  * @panHandle
75  * A reference returned by the Payment Application to refer to the PAN in question. Commerce Application may choose to use panHandle to request an encrypted PAN via API CP_APP_REQUESTS_ENCRYPTED_CARD. Ex: 9898987673
76  */
77  private String panHandle;
78 
79  /**
80  * @panLast4
81  * The last four digits of the card holder PAN. Ex: 7767
82  */
83  private String panLast4;
84 
85  /**
86  * @cardCurrency
87  * Present only with EMV cards and represents EMV Tag9F42
88  */
89  private String cardCurrency;
90 
91  /**
92  * @cardCountry
93  * Present only with EMV cards and represents EMV Tag5F28
94  */
95  private String cardCountry;
96 
97  /**
98  * @cardPreferredLanguages
99  * Present only with EMV cards and represents EMV Tag5F2D
100  */
101  private String cardPreferredLanguages;
102 
103  /**
104  * @cardType
105  * Specifies card type used to obtain card data. Must be one of: MAG_STRIPE, CHIP, CTLS_CARD, CTLS_PHONE, MANUAL
106  */
107  private String cardType;
108 
109  /**
110  * @authCode
111  * The transaction Authorization Code usually provided by the payment processing service or acquirer. Ex: "765945"
112  */
113  private String authCode;
114 
115  /**
116  * @authResult
117  * Result of transaction authorization. Must be one of: AUTHORIZED_ONLINE, AUTHORIZED_OFFLINE, REJECTED_ONLINE, ABORTED
118  */
119  private String authResult;
120 
121  /**
122  * @panFirst2
123  * The first two digits of the card. The POS application can identify the card type based on this (TBD)
124  */
125  private String panFirst2;
126 
127  /**
128  * @receiptData
129  * The receipt HTML for printing, including all of the items added by various CP Apps. The POS may further wrap this receipt with items like the cashier name, store logo, or other customizations.
130  */
131  private String receiptData;
132 
133  /**
134  * @signature
135  * An array of bytes that represent the customer’s signature (TBD)
136  */
137  private String signature;
138 
139  protected PaymentStatus(Parcel in) {
140  transId = in.readString();
141  holderAuth = in.readString();
142  invoice = in.readString();
143  storeNum = in.readString();
144  lane = in.readString();
145  cashierId = in.readString();
146  transactionType = in.readString();
147  transAmount = in.readString();
148  panHandle = in.readString();
149  panLast4 = in.readString();
150  cardCurrency = in.readString();
151  cardCountry = in.readString();
152  cardPreferredLanguages = in.readString();
153  cardType = in.readString();
154  authCode = in.readString();
155  authResult = in.readString();
156  panFirst2 = in.readString();
157  receiptData = in.readString();
158  signature = in.readString();
159  }
160 
161  public static final Creator<PaymentStatus> CREATOR = new Creator<PaymentStatus>() {
162  @Override
163  public PaymentStatus createFromParcel(Parcel in) {
164  return new PaymentStatus(in);
165  }
166 
167  @Override
168  public PaymentStatus[] newArray(int size) {
169  return new PaymentStatus[size];
170  }
171  };
172 
173  @Override
174  public int describeContents() {
175  return 0;
176  }
177 
178  @Override
179  public void writeToParcel(Parcel dest, int flags) {
180  dest.writeString(transId);
181  dest.writeString(holderAuth);
182  dest.writeString(invoice);
183  dest.writeString(storeNum);
184  dest.writeString(lane);
185  dest.writeString(cashierId);
186  dest.writeString(transactionType);
187  dest.writeString(transAmount);
188  dest.writeString(panHandle);
189  dest.writeString(panLast4);
190  dest.writeString(cardCurrency);
191  dest.writeString(cardCountry);
192  dest.writeString(cardPreferredLanguages);
193  dest.writeString(cardType);
194  dest.writeString(authCode);
195  dest.writeString(authResult);
196  dest.writeString(panFirst2);
197  dest.writeString(receiptData);
198  dest.writeString(signature);
199  }
200 }
