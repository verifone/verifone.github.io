14 /**
15  * A utility that provides some basic reflection-based parcel building. Subclasses are expected
16  * to implement a constructor that receives a parcelable, and must call super when writing to
17  * the parcel. The {@link ParcelCreator} class should be used for the CREATOR object, this allows
18  * the object to be built properly from the parcel. This solves those problems associated with
19  * reconstructing objects that are sub-classes of the object defined in the AIDL.
20  */
21 public abstract class BaseParcel implements Parcelable {

67  /**
68  * Looks in the parcel for the specific, known fields that can let us load the creator by
69  * reflection.
70  */
71  @SuppressWarnings("unchecked")
72  @Nullable
73  private E dynamicallyLoadParcelCreator(Parcel source) {
74  // Get the class name from the parcel.
75  final int originalPosition = source.dataPosition();
76  final String classInitializerString = source.readString();
77  final String sourceClassName = getClassNameFromSourceString(classInitializerString);
78  final int sourceClassVersion = getParcelVersionFromSourceString(classInitializerString);
79  try {
80  // Use the Java reflection abilities to load the appropriate parcel
81  // creator and instantiate the relevant object.
82  Class ceClass = Class.forName(sourceClassName);
83  if (BaseParcel.class.isAssignableFrom(ceClass)) {
84  Class<? extends BaseParcel> statusClass = (Class<? extends BaseParcel>) ceClass;
85  // get parcel constructor here.
86  try {
87  Constructor<? extends BaseParcel> constructor =
88  statusClass.getDeclaredConstructor(Parcel.class, int.class);
89  return (E) constructor.newInstance(source, sourceClassVersion);
90  } catch (NoSuchMethodException | InstantiationException |
91  InvocationTargetException |
92  IllegalAccessException |
93  IllegalArgumentException |
94  ExceptionInInitializerError e) {
95  Log.w(TAG, "Unable to use constructor for createdObject " +
96  sourceClassName + " with version " + sourceClassVersion + ".", e);
97  }
98 
99  source.setDataPosition(originalPosition);
100  Field creatorField = statusClass.getDeclaredField("CREATOR");
101  Class creatorClass = creatorField.getType();
102  if (Parcelable.Creator.class.isAssignableFrom(creatorClass)) {
103  Parcelable.Creator actualCreator =
104  (Parcelable.Creator) creatorField.get(null);
105  E createdObject = (E) actualCreator.createFromParcel(source);
106  Log.d(TAG, "createdObject by Parcelable " +
107  createdObject.getClass().getName());
108  return createdObject;
109  }
110  }
111  Log.d(TAG, "Unable to load class " + sourceClassName + " with version " +
112  sourceClassVersion + " dynamically.");
113  } catch (ClassNotFoundException |
114  NoSuchFieldException |
115  IllegalAccessException e) {
116  Log.w(TAG, "Error getting values from class " + sourceClassName
117  + " with version " + sourceClassVersion + ".", e);
118  }
119  source.setDataPosition(originalPosition);
120  return null;
121  }
122  }

124  private static String getClassNameFromSourceString(String source) {
125  String className = source;
126  if (source != null) {
127  final int separatorIndex = source.lastIndexOf(":");
128  if (separatorIndex > 0)
129  className = source.substring(0, separatorIndex);
130  }
131  return className;
132  }
133 
134  private static int getParcelVersionFromSourceString(String source) {
135  int version = 0;
136  if (source != null) {
137  final int separatorIndex = source.lastIndexOf(":") + 1;
138  if (separatorIndex > 0 && separatorIndex < source.length()) {
139  String versionStr = source.substring(separatorIndex);
140  try {
141  version = Integer.parseInt(versionStr);
142  } catch (NumberFormatException e) { /* Suppressed. */ }
143  }
144  }
145  return version;
146  }
