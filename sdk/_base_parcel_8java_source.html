89  /**
90  * Looks in the parcel for the specific, known fields that can let us load the creator by
91  * reflection.
92  */
93  @SuppressWarnings("unchecked")
94  @Nullable
95  private E dynamicallyLoadParcelCreator(Parcel source) {
96  // Get the class name from the parcel.
97  final int originalPosition = source.dataPosition();
98  final String classInitializerString = source.readString();
99  final String sourceClassName = getClassNameFromSourceString(classInitializerString);
100  if (sourceClassName != null) {
101  final int sourceClassVersion = getParcelVersionFromSourceString(classInitializerString);
102  mRecommendedParcelVersion = sourceClassVersion;
103  try {
104  // Use the Java reflection abilities to load the appropriate parcel
105  // creator and instantiate the relevant object.
106  Class ceClass = Class.forName(sourceClassName);
107  if (BaseParcel.class.isAssignableFrom(ceClass)) {
108  Class<? extends BaseParcel> statusClass
109  = (Class<? extends BaseParcel>) ceClass;
110  // get parcel constructor here.
111  try {
112  Constructor<? extends BaseParcel> constructor =
113  statusClass.getDeclaredConstructor(Parcel.class, int.class);
114  return (E) constructor.newInstance(source, sourceClassVersion);
115  } catch (NoSuchMethodException | InstantiationException |
116  InvocationTargetException |
117  IllegalAccessException |
118  IllegalArgumentException |
119  ExceptionInInitializerError e) {
120  Log.w(TAG, "Unable to use constructor for createdObject " +
121  sourceClassName + " with version " + sourceClassVersion +
122  ".", e);
123  }
124 
125  source.setDataPosition(originalPosition);
126  Field creatorField = statusClass.getDeclaredField("CREATOR");
127  Class creatorClass = creatorField.getType();
128  if (Parcelable.Creator.class.isAssignableFrom(creatorClass)) {
129  Parcelable.Creator actualCreator =
130  (Parcelable.Creator) creatorField.get(null);
131  E createdObject = (E) actualCreator.createFromParcel(source);
132  Log.d(TAG, "createdObject by Parcelable " +
133  createdObject.getClass().getName());
134  return createdObject;
135  }
136  }
137  Log.d(TAG, "Unable to load class " + sourceClassName + " with version " +
138  sourceClassVersion + " dynamically.");
139  } catch (ClassNotFoundException |
140  NoSuchFieldException |
141  IllegalAccessException e) {
142  Log.w(TAG, "Error getting values from class " + sourceClassName
143  + " with version " + sourceClassVersion + ".", e);
144  }
145  } else {
146  mRecommendedParcelVersion = -1;
147  }
148  source.setDataPosition(originalPosition);
149  return null;
150  }
151  }

188  private static int getParcelVersionFromSourceString(String source) {
189  int version = 0;
190  if (source != null) {
191  final int separatorIndex = source.lastIndexOf(":") + 1;
192  if (separatorIndex > 0 && separatorIndex < source.length()) {
