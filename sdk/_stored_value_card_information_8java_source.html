118  public StoredValueCardInformation(Parcel in, int parcelVersion) {
119  super(in, parcelVersion);
120  in.readString();
121  this.mProvider = in.readString();
122  this.mType = in.readString();
123  this.mPinCode = in.readString();
124  this.mCvv2 = in.readString();
125  this.mBarcode = in.readString();
126  if (in.dataAvail() > 0) {
127  this.mEbtType = ConversionUtility.readEnumFromParcel(in, EbtType.class);
128  }
129  }

131  @Override
132  public void writeToParcel(Parcel dest, int flags) {
133  super.writeToParcel(dest, flags);
134  ConversionUtility.writeBigDecimalToParcel(this.getAvailableBalance(), dest);
135  dest.writeString(this.mProvider);
136  dest.writeString(this.mType);
137  dest.writeString(this.mPinCode);
138  dest.writeString(this.mCvv2);
139  dest.writeString(this.mBarcode);
140  ConversionUtility.writeEnumToParcel(dest, mEbtType);
141  }
