147  public StoredValueCardInformation(Parcel in, int parcelVersion) {
148  super(in, parcelVersion);
149  in.readString();
150  this.mProvider = in.readString();
151  this.mType = in.readString();
152  this.mPinCode = in.readString();
153  this.mCvv2 = in.readString();
154  this.mBarcode = in.readString();
155  if (in.dataAvail() > 0) {
156  this.mEbtType = ConversionUtility.readEnumFromParcel(in, EbtType.class);
157  }
158  }

170  }
